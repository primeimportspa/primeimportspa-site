Ejemplo: Lectura de sensor con ESP32

#include "DHT.h"
#define DHTPIN 4
#define DHTTYPE DHT22

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(115200);
  dht.begin();
}

void loop() {
  float h = dht.readHumidity();
  float t = dht.readTemperature();
  if (!isnan(h) && !isnan(t)) {
    Serial.print("Humedad: "); Serial.print(h);
    Serial.print("% Temp: "); Serial.println(t);
  }
  delay(2000);
}
